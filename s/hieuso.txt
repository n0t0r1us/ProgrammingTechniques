#include <iostream>
#include <vector>
using namespace std;
int main() {
    int n;
    cin >> n;
    vector<int> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    int maxDifference = a[1] - a[0];
    int minElement = a[0];
    for (int j = 1; j < n; j++) {
        int currentDifference = a[j] - minElement;
        if (currentDifference > maxDifference) {
            maxDifference = currentDifference;
        }
        if (a[j] < minElement) {
            minElement = a[j];
        }
    }
    cout << maxDifference << endl;
    return 0;
}
