#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

int main() {
    int n, k;
    cin >> n >> k;

    vector<int> floors(n);
    for (int i = 0; i < n; ++i) {
        cin >> floors[i];
    }
    sort(floors.rbegin(), floors.rend());

    int totalTime = 0;
    for (int i = 0; i < n; i += k) {
        int highestFloorInGroup = floors[i];
        int timeInGroup = 2 * (highestFloorInGroup - 1);

        totalTime += timeInGroup;
    }

    cout << totalTime << endl;

    return 0;
}
