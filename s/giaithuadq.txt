#include <iostream>

long long factorial(int n) {
    if (n == 0 || n == 1) {
        return 1;
    } else {
        return n * factorial(n - 1);
    }
}

int main() {
    int n;
    std::cin >> n;

    if (n < 1 || n > 17) {
        return 1;
    }

    long long result = factorial(n);

    std::cout << result << std::endl;

    return 0;
}
