#include <iostream>
bool isPowerOfTwo(int num) {
    if (num <= 0 || (num & (num - 1)) != 0) {
        return false;
    }
    return true;
}
int main() {
    int n;
    std::cin >> n;
    for (int i = 0; i < n; i++) {
        int a;
        std::cin >> a;
        
        if (isPowerOfTwo(a)) {
            std::cout << "1" << std::endl;
        } else {
            std::cout << "0" << std::endl;
        }
    }
    return 0;
}
