#include <iostream>
#include <vector>
using namespace std;
int main() {
    int n, m;
    cin >> n >> m;
    vector<int> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    for (int i = 0; i < n; i++) {
        int low = i + 1; 
        int high = n;
        while (low < high) {
            int mid = (low + high) / 2; 
            if (a[mid] - a[i] == m) {
                cout << a[i] << " " << a[mid] << endl;
                return 0;
            } else if (a[mid] - a[i] < m) {
                low = mid + 1;
            } else {
                high = mid;
            }
        }
    }
    cout << -1 << endl; 
    return 0;
}
