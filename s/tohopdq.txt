#include <iostream>
long long combination(int k, int n) {
    if (k == 0 || k == n) {
        return 1;
    } else {
        return combination(k, n - 1) + combination(k - 1, n - 1);
    }
}

int main() {
    int n, k;
    std::cin >> n >> k;

    if (n < 1 || n > 50 || k < 0 || k > n) {
        std::cout << "" << std::endl;
        return 1;
    }

    long long result = combination(k, n);

    std::cout << result << std::endl;

    return 0;
}
