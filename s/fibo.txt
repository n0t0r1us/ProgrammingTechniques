#include <iostream>
using namespace std;
const int MOD = 1000000;
int main() {
    int n;
    cin >> n;
    if (n <= 2) {
        cout << "1" << endl;
        return 0;
    }
    int fn_2 = 1; 
    int fn_1 = 1;
    int fn = 0; 
    for (int i = 3; i <= n; ++i) {
        fn = (fn_1 + fn_2) % MOD;
        fn_2 = fn_1;
        fn_1 = fn;
    }
    cout << fn << endl;
    return 0;
}
