#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
int main() {
    int n;
    cin >> n;
    vector<int> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    sort(a.begin(), a.end(), greater<int>());
    int maxMilk = 0;
    for (int i = 0; i < n; ++i) {
        int milk = a[i] - i;
        if (milk > 0) {
            maxMilk += milk;
        }
    }
    cout << maxMilk << endl;
    return 0;
}
