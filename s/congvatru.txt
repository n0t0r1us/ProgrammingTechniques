#include <iostream>
#include <algorithm>

using namespace std;

int main() {
    int n, k;
    cin >> n >> k;

    int x;
    cin >> x;

    int a[n - 1];
    for (int i = 0; i < n - 1; ++i) {
        cin >> a[i];
    }
    sort(a, a + n - 1);

    int result = x;
    for (int i = 0; i < n - 1; ++i) {
        if (i < k) {
            result -= a[i];
        } else {
            result += a[i];
        }
    }

    cout << result << endl;

    return 0;
}
