#include <iostream>
#include <vector>

using namespace std;

long long countWaysToTile(int n) {
    vector<long long> dp(n + 1, 0);

    dp[0] = 1;

    for (int i = 1; i <= n; ++i) {
        dp[i] = dp[i - 1]; 

        if (i >= 2) {
            dp[i] += dp[i - 2]; 
        }
    }

    return dp[n];
}

int main() {
    int n;
    cin >> n;

    long long result = countWaysToTile(n);

    cout << result << endl;

    return 0;
}
