#include <iostream>
#include <vector>
using namespace std;
int main() {
    int n;
    cin >> n;
    vector<int> distances(n - 1); 
    vector<int> fuel_prices(n);
    for (int i = 0; i < n - 1; i++) {
        cin >> distances[i];
    }
    for (int i = 0; i < n; i++) {
        cin >> fuel_prices[i];
    }
    long long total_cost = 0;
    int min_fuel_price = fuel_prices[0];
    for (int i = 0; i < n - 1; i++) {
        min_fuel_price = min(min_fuel_price, fuel_prices[i]);
        total_cost += (long long)min_fuel_price * distances[i];
    }
    cout << total_cost << endl;
    return 0;
}
