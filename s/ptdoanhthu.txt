#include <iostream>
#include <climits>

int doanhthu(int a[100], int n) {
    int max_ending_here = 0;
    int max_so_far = INT_MIN;

    for (int i = 0; i < n; i++) {
        max_ending_here = std::max(max_ending_here + a[i], a[i]);
        max_so_far = std::max(max_so_far, max_ending_here);
    }

    return max_so_far;
}

int main() {
    int n;
    std::cin >> n;

    int a[100];
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }

    std::cout << doanhthu(a, n) << std::endl;

    return 0;
}
